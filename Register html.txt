    <script>
        const form = document.getElementById('registrationForm');
        const statusMessage = document.getElementById('statusMessage');
        const formUrl = 'https://docs.google.com/forms/d/e/1FAIpQLSd-kqNgGsxB2g0tzBjFU7GQkUwVQkunK4Z1XA07JsQ8vYIRuA/formResponse';

        form.addEventListener('submit', async (event) => {
            event.preventDefault();

            const fullName = document.querySelector('input[name="entry.287163873"]').value;
            const phone = document.querySelector('input[name="entry.1929394666"]').value;

            if (!fullName || !phone) {
                statusMessage.className = 'error';
                statusMessage.textContent = 'กรุณากรอกข้อมูลในช่องที่จำเป็นให้ครบถ้วน';
                return;
            }

            statusMessage.className = '';
            statusMessage.textContent = 'กำลังส่งข้อมูล...';

            const formData = new FormData(form);
            const params = new URLSearchParams();

            const checkedCarType1 = document.querySelector('input[name="entry.1528054360"]:checked');
            if (checkedCarType1) {
                params.append(checkedCarType1.name, checkedCarType1.value);
            }

            const checkedCarType2 = document.querySelector('input[name="entry.839458641"]:checked');
            if (checkedCarType2) {
                params.append(checkedCarType2.name, checkedCarType2.value);
            }
            
            const checkedTransportType = document.querySelector('input[name="entry.895767174"]:checked');
            if (checkedTransportType) {
                params.append(checkedTransportType.name, checkedTransportType.value);
            }
            
            params.append('entry.287163873', formData.get('entry.287163873'));
            params.append('entry.1929394666', formData.get('entry.1929394666'));
            params.append('entry.1641413677', formData.get('entry.1641413677'));

            try {
                const response = await fetch(formUrl, {
                    method: 'POST',
                    body: params,
                    mode: 'no-cors'
                });
                
                statusMessage.className = 'success';
                statusMessage.textContent = 'การสมัครสำเร็จแล้ว! ขอบคุณที่เข้าร่วม';
                form.reset();
            } catch (error) {
                console.error('Submission error:', error);
                statusMessage.className = 'error';
                statusMessage.textContent = 'เกิดข้อผิดพลาดในการส่งข้อมูล กรุณาลองอีกครั้ง';
            }
        });
    </script>
